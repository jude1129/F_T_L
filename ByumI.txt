import 'package:flutter/material.dart';

void main() {
  runApp(const BMIApp());
}

class BMIApp extends StatelessWidget {
  const BMIApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'BMI Calculator',
      theme: ThemeData(
        colorScheme: ColorScheme.fromSeed(seedColor: Colors.teal),
        useMaterial3: true,
      ),
      home: const BMICalculatorPage(),
    );
  }
}

class BMICalculatorPage extends StatefulWidget {
  const BMICalculatorPage({super.key});

  @override
  State<BMICalculatorPage> createState() => _BMICalculatorPageState();
}

class _BMICalculatorPageState extends State<BMICalculatorPage> {
  final _weightController = TextEditingController();
  final _heightController = TextEditingController();
  final _formKey = GlobalKey<FormState>();

  double? _bmi;
  String? _category;
  String? _message;

  @override
  void dispose() {
    _weightController.dispose();
    _heightController.dispose();
    super.dispose();
  }

  void _calculateBMI() {
    if (!_formKey.currentState!.validate()) return;

    final weight = double.parse(_weightController.text.trim()); // kg
    final heightCm = double.parse(_heightController.text.trim()); // cm
    final heightM = heightCm / 100.0;

    final bmi = weight / (heightM * heightM);

    final category = _bmiCategory(bmi);
    final msg = _bmiMessage(category);

    setState(() {
      _bmi = double.parse(bmi.toStringAsFixed(1));
      _category = category;
      _message = msg;
    });
  }

  String _bmiCategory(double bmi) {
    if (bmi < 18.5) return "Underweight";
    if (bmi < 25) return "Normal";
    if (bmi < 30) return "Overweight";
    return "Obesity";
  }

  String _bmiMessage(String category) {
    switch (category) {
      case "Underweight":
        return "Your BMI is lower than the typical range.";
      case "Normal":
        return "Your BMI is within the typical range.";
      case "Overweight":
        return "Your BMI is higher than the typical range.";
      case "Obesity":
        return "Your BMI is significantly above the typical range.";
      default:
        return "";
    }
  }

  Color _categoryColor(String? category) {
    switch (category) {
      case "Underweight":
        return Colors.blueAccent;
      case "Normal":
        return Colors.green;
      case "Overweight":
        return Colors.orange;
      case "Obesity":
        return Colors.red;
      default:
        return Colors.grey;
    }
  }

  void _reset() {
    _weightController.clear();
    _heightController.clear();
    setState(() {
      _bmi = null;
      _category = null;
      _message = null;
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('BMI Calculator')),
      body: SafeArea(
        child: SingleChildScrollView(
          padding: const EdgeInsets.all(16),
          child: Form(
            key: _formKey,
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.stretch,
              children: [
                const Text(
                  "Enter your details",
                  style: TextStyle(fontSize: 20, fontWeight: FontWeight.w600),
                ),
                const SizedBox(height: 12),
                TextFormField(
                  controller: _weightController,
                  keyboardType: TextInputType.numberWithOptions(decimal: true),
                  decoration: const InputDecoration(
                    labelText: 'Weight (kg)',
                    hintText: 'e.g., 70.5',
                    border: OutlineInputBorder(),
                  ),
                  validator: (value) {
                    if (value == null || value.trim().isEmpty) {
                      return 'Please enter your weight in kg';
                    }
                    final w = double.tryParse(value.trim());
                    if (w == null || w <= 0 || w > 400) {
                      return 'Enter a valid weight (0–400 kg)';
                    }
                    return null;
                  },
                ),
                const SizedBox(height: 12),
                TextFormField(
                  controller: _heightController,
                  keyboardType: TextInputType.numberWithOptions(decimal: true),
                  decoration: const InputDecoration(
                    labelText: 'Height (cm)',
                    hintText: 'e.g., 175',
                    border: OutlineInputBorder(),
                  ),
                  validator: (value) {
                    if (value == null || value.trim().isEmpty) {
                      return 'Please enter your height in cm';
                    }
                    final h = double.tryParse(value.trim());
                    if (h == null || h <= 0 || h > 300) {
                      return 'Enter a valid height (0–300 cm)';
                    }
                    return null;
                  },
                ),
                const SizedBox(height: 16),
                Row(
                  children: [
                    Expanded(
                      child: ElevatedButton.icon(
                        onPressed: _calculateBMI,
                        icon: const Icon(Icons.calculate),
                        label: const Text('Calculate'),
                      ),
                    ),
                    const SizedBox(width: 12),
                    OutlinedButton(
                      onPressed: _reset,
                      child: const Text('Reset'),
                    ),
                  ],
                ),
                const SizedBox(height: 24),
                if (_bmi != null) ...[
                  Card(
                    elevation: 1,
                    child: Padding(
                      padding: const EdgeInsets.all(16),
                      child: Column(
                        crossAxisAlignment: CrossAxisAlignment.stretch,
                        children: [
                          Text(
                            'Your BMI: $_bmi',
                            style: const TextStyle(
                              fontSize: 24,
                              fontWeight: FontWeight.bold,
                            ),
                          ),
                          const SizedBox(height: 8),
                          Row(
                            children: [
                              Container(
                                width: 12,
                                height: 12,
                                decoration: BoxDecoration(
                                  color: _categoryColor(_category),
                                  shape: BoxShape.circle,
                                ),
                              ),
                              const SizedBox(width: 8),
                              Text(
                                _category ?? '',
                                style: TextStyle(
                                  fontSize: 18,
                                  fontWeight: FontWeight.w600,
                                  color: _categoryColor(_category),
                                ),
                              ),
                            ],
                          ),
                          const SizedBox(height: 8),
                          Text(
                            _message ?? '',
                            style: const TextStyle(fontSize: 16),
                          ),
                        ],
                      ),
                    ),
                  ),
                  const SizedBox(height: 12),
                  // Optional: a simple visual scale
                  const Text(
                    'BMI categories (for reference)',
                    style: TextStyle(fontWeight: FontWeight.w600),
                  ),
                  const SizedBox(height: 8),
                  Wrap(
                    spacing: 8,
                    runSpacing: 8,
                    children: const [
                      Chip(label: Text('< 18.5 Underweight'), backgroundColor: Color(0xFFBBDEFB)),
                      Chip(label: Text('18.5–24.9 Normal'), backgroundColor: Color(0xFFC8E6C9)),
                      Chip(label: Text('25–29.9 Overweight'), backgroundColor: Color(0xFFFFE0B2)),
                      Chip(label: Text('≥ 30 Obesity'), backgroundColor: Color(0xFFFFCDD2)),
                    ],
                  ),
                ],
              ],
            ),
          ),
        ),
      ),
    );
  }
}
